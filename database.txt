create table login
(
	ID int auto_increment
		primary key,
	username varchar(50) not null,
	password varchar(255) not null,
	roli text not null,
	constraint login_password_uindex
		unique (password),
	constraint login_roli_uindex
		unique (roli) using hash,
	constraint login_username_uindex
		unique (username)
);

create table posts
(
	image_url text null,
	author text null,
	dita text null,
	headLine text null,
	description text null
);
